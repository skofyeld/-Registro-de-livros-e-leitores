Tabela sobre registro de livros e leitores no PostgreSQL.



CREATE TABLE public.leitor (
    	Codigo integer,
    	Leitor character varying,
	Telefone integer,
	CPF integer,
		CONSTRAINT leitor_pkey PRIMARY KEY (codigo)
);
CREATE TABLE public.autor (
    	Codigo integer,
    	Autor character varying,
		CONSTRAINT autor_pkey PRIMARY KEY (codigo)
);
CREATE TABLE public.livros (
	Codigo integer,
    	Titulo character varying,
	Editora character varying,
	Assunto character varying,
		CONSTRAINT cliente_pkey PRIMARY KEY (codigo)
);
CREATE TABLE public.assunto (
	Codigo integer,
	Sigla character varying,
    	Descrição character varying,
		CONSTRAINT assunto_pkey PRIMARY KEY (codigo)
);
CREATE TABLE public.editora (
    	Codigo integer,
    	Nome character varying,
		CONSTRAINT editora_pkey PRIMARY KEY (codigo)
);
